def set_listtype_options
  @listtype = params[:listtype] || 'word'
  @listtype_options = []
  ['word', 'lemma', 'pos', 'phonetic'].each do |type|
    @listtype_options << [t(:"data_labels.keys.#{type}").capitalize, type]
  end
end
