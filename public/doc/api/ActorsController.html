def show_key
  if !@admin_logged_in
    redirect_to 'admin/login'
  end
  if @key
    data = get_actor_key_values(@key, 0, 0, "actor_count", "desc")
    @value_count = data['values'].length
    if @value_count > 5
      @values = data['values'][0..4]
    else
      @values = data['values']
    end
  end
end
